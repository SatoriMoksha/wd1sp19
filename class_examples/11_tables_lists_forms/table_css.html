CSS

table{
	border-collapse: collapse;
}
th,
td{
	padding: .25em .5em;
	border: 1px solid #000000;
}
.left{
	text-align: left;
}
thead{
	border-bottom: 3px solid #000000;
}
tfoot{
	border-top: 3px solid #000000;
}
